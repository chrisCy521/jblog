var cMap; // 地图
    var cSnake; // 蛇
    var cFood; // 食物
    var score = 0; // 分数
    var speed = 600; // 蛇前进的速度

    var game = (function () { // 单例模式开发
      // 创建地图的构造函数
      function Gmap() {
        this._map = null;
        this.createmap = function () { // 创建地图
          if (!this._map) {
            this._map = document.createElement('div')
            this._map.className = 'map'
          }
        }
        this.createSpan = function () { // 创建分数显示栏
          this._span = null
          if (!this._span) {
            this._span = document.createElement("span")
            this._span.className = "score"
            this._span.innerHTML = '总分:0'
          }
          this._map.appendChild(this._span)
        }
      }


      // 创建食物的构造函数
      function Food() {
        this._food = null
        // 因为食物是在地图上创建的 所以说我们需要地图
        this._map = null
        this.x = 0; //坐标
        this.y = 0;
        this.createFood = function () {
          this.x = Math.floor(Math.random() * 30)
          this.y = Math.floor(Math.random() * 20)
          if (!this._food) {
            this._food = document.createElement("div")
            this._food.className = 'food'
          }

          this._food.style.left = this.x * 30 + 'px'
          this._food.style.top = this.y * 30 + 'px'
          // 食物是地图的子集
          this._map.appendChild(this._food)
        }


      }

      // 关于蛇的构造函数
      function Snake() {
        this.direct = 'right' // 默认的移动方向
        this._map = null // 地图
        this._span = null // 分数
        this._snake = [ // 初始蛇 有三块 头 身 尾  利用数组存储相对坐标 颜色和判断值 
          [3, 1, 'skyblue', null],
          [2, 1, 'red', null],
          [1, 1, 'red', null],
        ]
        // 创建蛇
        this.createSnake = function () {
          // 循环数组
          for (var i = 0; i < this._snake.length; i++) {
            if (!this._snake[i][3]) {
              this._snake[i][3] = document.createElement('div')
              this._snake[i][3].style.backgroundColor = this._snake[i][2]
              this._snake[i][3].className = 'snake'
            }
            // 地图大小是900 * 600 蛇一节是30*30 推的结果
            this._snake[i][3].style.left = this._snake[i][0] * 30 + 'px'
            this._snake[i][3].style.top = this._snake[i][1] * 30 + 'px'
            this._map.appendChild(this._snake[i][3]) // 蛇也是地图的子集
          }
        }

        // 蛇的移动
        this.snake_move = function () { // 蛇移动 其实就是坐标的迁移
          // 倒着循环遍历 第二节 覆盖 第三节 第一节 覆盖 第二节  前面一节 移动
          var len = this._snake.length - 1;
          for (var i = len; i > 0; i--) {
            // 后面的覆盖前面的
            this._snake[i][1] = this._snake[i - 1][1]
            this._snake[i][0] = this._snake[i - 1][0]
          }
          switch (this.direct) {//移动方向
            case 'right':
              this._snake[0][0] += 1;
              break;
            case 'left':
              this._snake[0][0] -= 1;
              break;

            case 'top':
              this._snake[0][1] += 1;
              break;

            case 'down':
              this._snake[0][1] -= 1;
              break;
          }
          // 在移动的过程中 判断是否吃到了食物 
          //判断利用蛇头相对坐标和食物相对坐标是否吃到食物
          if (this._snake[0][0] == cFood.x && this._snake[0][1] == cFood.y) {
            score += 100 //吃到食物加分 满500分加速一次
            this._span.innerHTML = `总分${score}分`
            if (score % 500 == 0 && speed > 100) {
              speed -= 100
              // 清除以前的移动 重新移动
              clearInterval(timecount)
              setInterval(function () {
                cSnake.snake_move()
              }, speed)
            }

            // 增加蛇长
            this._snake.push([
              this._snake[this._snake.length - 1][0],
              this._snake[this._snake.length - 1][1],
              "yellow",
              null
            ])
            // 创建食物
            cFood.createFood()
          }

          // 判断是否碰到了墙壁
          if (this._snake[0][0] * 30 < 0 || this._snake[0][0] * 30 > 870 || this._snake[0][1] *
            30 < 0 || this._snake[0][1] * 30 > 570) {//如果碰到墙壁，游戏结束
            clearInterval(timecount);
            this._span.innerHTML = 'GAME OVER';
          }
          // 判断是否吃到了自己
          for (var i = 0; i < this._snake.length; i++) {
            if (i > 0 && this._snake[0][1] == this._snake[i][1] && this._snake[0][0] == this._snake[i][0]) {
              clearInterval(timecount);
              this._span.innerHTML = 'GAME OVER';
            }
          }
          this.createSnake();
        }
      }


      function createFood(map) {
        cFood = new Food
        cFood._map = map
        cFood.createFood()
      }

      function createSnake(map, span) {
        cSnake = new Snake()
        cSnake._map = map
        cSnake._span = span
        cSnake.createSnake()
        // 通过定时器 让蛇移动起来
        timecount = setInterval(function () {
          cSnake.snake_move();
        }, speed);

        // 键盘的移动 控制蛇移动的方向
        document.onkeypress = function (e) {//wasd  来控制蛇的方向（不能直接转换到相反方向）
          switch (e.key) {
            case 's':
              if (cSnake.direct == 'down') {
                return;
              }
              cSnake.direct = 'top';
              break;
            case 'a':
              if (cSnake.direct == 'right') {
                return;
              }
              cSnake.direct = 'left';
              break;
            case 'w':
              if (cSnake.direct == 'top') {
                return;
              }
              cSnake.direct = 'down';
              break;
            case 'd':
              if (cSnake.direct == 'left') {
                return;
              }
              cSnake.direct = 'right';
              break;
          }
        }

      }
      // 向外暴露一个方法
      function getMaps() {
        cMap = new Gmap() // 实例化对象
        cMap.createmap() // 创建地图
        cMap.createSpan() // 创建分数
        // 创建食物 定义一个方法 要用到地图 把地图传过去
        createFood(cMap._map)
        createSnake(cMap._map, cMap._span)
        return cMap._map
      }

      return {
        createGame: getMaps()
      }

    })();


    document.body.appendChild(game.createGame)
